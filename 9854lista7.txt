#include<stdio.h>

int funcao (int mat[5][5]){

    int soma = 0;
    int i;

    for(i = 0; i<5; i++){
        soma += mat[i][4-i];
    }

    return soma;
}

int main (void){

    int mat [5][5] = {
    {0,0,0,0,1},
    {0,0,0,1,0},
    {0,0,1,0,0},
    {0,1,0,0,0},
    {1,0,0,0,0}
    };

    int soma_inversa = funcao(mat);

    printf("A soma da diagonal inversa eh: %d", soma_inversa);

    return 0;
}
